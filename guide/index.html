# 快速开始

# 安装组件库

npm i tu-view

# 引用组件库

在 main.js 中引用组件库

// 全部引入
import TuView from "tu-view";
import "tu-view/lib/tuview.css";
Vue.use(TuView);

// 按需引入
import "tu-view/lib/packages/button/style.css";
import { Button } from "tu-view";
Vue.use(Button);
